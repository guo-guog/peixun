create or replace procedure wcs_getinboundtaskpy_p(i_warehouseid        IN wcs_taskinfo.warehouseid%TYPE, --仓库编号
                                                   i_barcode            IN VARCHAR2, --托盘条码
                                                   i_fromstation        IN wcs_taskinfo.fromstation%TYPE, --起始地址
                                                   i_userid             IN VARCHAR2, --用户编码
                                                   i_insertwmsflag      IN INTEGER, --是否需要产生WMS入库任务
                                                   i_insertstvflag      IN NUMBER, --是否需要写入穿梭车任务
                                                   i_insertitemcodeflag IN NUMBER, --是否需要写入一号工程条码
                                                   i_emptypalletinbound IN NUMBER, --是否是空托盘组入库
                                                   i_taskno             IN INTEGER, --任务号
                                                   i_checkflag          IN INTEGER, --是否通过货型检测和条码读取
                                                   i_itemcodes          IN VARCHAR2, --物料条码集,
                                                   o_result             OUT INTEGER, --执行成功标志
                                                   o_taskno             OUT INTEGER, --返回的任务号
                                                   o_tostation          OUT VARCHAR2 --返回的目标站台
                                                   ) AS
  v_wmstaskno         task.tasknumber%TYPE;
  v_tasktypecode      task.tasktypecode%TYPE;
  v_equipmentnumber   equipment.equipmentnumber%TYPE;
  v_sourceaddressid   task.sourceaddressid%TYPE;
  v_sourceaddresstype task.sourceaddresstype%TYPE;
  v_sourceaddresstext task.sourceaddresstext%TYPE;
  v_targetaddressid   task.targetaddressid%TYPE;
  v_targetaddresstype task.targetaddresstype%TYPE;
  v_targetaddresstext task.targetaddresstext%TYPE;
  v_ledtext           task.ledtext%TYPE;
  v_ledtext2          task.ledtext%TYPE;
  v_containerid       task.containerid%TYPE;
  v_containerbarcode  task.containerbarcode%TYPE;
  v_agvflag           task.lgvexist%TYPE;
  v_groupnumber       task.groupnumber%TYPE;
  v_itemid            task.itemid%TYPE;
  v_itemqty           task.operationqty%TYPE;
  v_priority          VARCHAR2(10);
  v_taskstatus        VARCHAR2(20);
  v_wmsorderno        VARCHAR2(50);
  v_wmsorderqty       NUMBER;
  v_wmsbatchno        VARCHAR2(50);
  v_wmssequenceno     VARCHAR2(50);
  v_itemno            VARCHAR2(50);
  v_itemname          VARCHAR2(50);
  v_taskno            NUMBER;
  v_toconvery         VARCHAR2(10);
  v_runstatus         VARCHAR2(10);
  v_recordcount       NUMBER; --查找数据库记录的数量
  v_result            NUMBER;
  o_inboundtask       sys_refcursor; --反馈的结果集
  v_cpequip           VARCHAR(20);
  v_rollbackflag      INTEGER;
  v_wmscount          INTEGER;
  v_taskqty           INTEGER;
begin
  v_wmscount  := 0;
  v_result    := 0;
  o_result    := 0;
  o_taskno    := 0;
  o_tostation := 0;
  if (i_checkflag = 0) then
    if (i_taskno < 10000) then
      wcs_gettaskno_p(i_warehouseid, v_taskno);
    else
      v_taskno := i_taskno;
    end if;
    if (i_insertwmsflag = 1 and i_emptypalletinbound = 1) then
      --空托盘组入库
      i_pallets_put(i_warehouseid, i_barcode, v_result, o_inboundtask);
    end if;
    if (i_insertwmsflag = 1 and i_emptypalletinbound = 0) then
      --片烟入库产生新任务
      i_directinbound_py(i_warehouseid, i_barcode, v_result, o_inboundtask);
    end if;
    if (i_insertwmsflag = 0) then
      --入库获取任务
      i_gettask_inbound(i_warehouseid,
                        i_barcode,
                        '',
                        v_rollbackflag,
                        o_inboundtask);
    end if;
  
    LOOP
      EXIT WHEN o_inboundtask%NOTFOUND;
      FETCH o_inboundtask
        INTO v_wmstaskno,
             v_tasktypecode,
             v_equipmentnumber,
             v_sourceaddressid,
             v_sourceaddresstype,
             v_sourceaddresstext,
             v_targetaddressid,
             v_targetaddresstype,
             v_targetaddresstext,
             v_ledtext,
             v_containerid,
             v_containerbarcode,
             v_agvflag,
             v_groupnumber,
             v_itemid,
             v_itemqty;
      v_wmscount := o_inboundtask%ROWCOUNT;
    END LOOP;
  
    IF (v_wmscount > 0) THEN
      wcs_insertinboundtask_p(i_warehouseid,
                              v_wmstaskno,
                              i_barcode,
                              i_fromstation,
                              v_targetaddresstext,
                              v_priority,
                              v_tasktypecode,
                              v_taskstatus,
                              v_ledtext,
                              v_wmsorderno,
                              v_wmsorderqty,
                              v_wmsbatchno,
                              v_groupnumber,
                              v_wmssequenceno,
                              v_itemno,
                              v_itemname,
                              v_itemqty,
                              i_userid,
                              v_taskno,
                              v_equipmentnumber,
                              v_result);
      --查询输送机目的地址
      IF (v_result = 1) THEN
        --写入WCS事件日志
        wcs_createeventlog_p(i_warehouseid,
                             'WCS获取WMS任务后，写入到WCS任务表成功，WMS任务号' ||
                             v_wmstaskno || 'WCS任务号' || v_wmstaskno);
      
        SELECT tostation
          INTO o_tostation
          FROM wcs_subtaskinfo
         WHERE taskno = v_taskno
           AND taskstatus = '0'
           AND equipmenttype = 'STA'
           AND ROWNUM < 2;
      
        o_taskno := v_taskno;
        --写入WCS事件日志
        wcs_createeventlog_p(i_warehouseid,
                             '入库，从WCS任务表中获取到入库任务,任务号:' || o_taskno ||
                             '起始地址:' || i_fromstation || '结束地址:' ||
                             o_tostation || '条码:' || i_barcode);
      end if;
    end if;
    IF (v_result = 1) THEN
      COMMIT;
      o_result := 1;
    ELSE
      --执行不成功，回滚任务
      ROLLBACK;
    
      IF (v_rollbackflag > 0 OR i_insertwmsflag = 1) --需要回滚货位信息
       THEN
        loc_updatelocstate_in_p(v_targetaddressid, 'WCS');
      END IF;
    END IF;
  else
    wcs_createeventlog_p(i_warehouseid,
                         '站台' || i_fromstation || '申请入库失败，外形检测故障');
  end if;
end wcs_getinboundtaskpy_p;
